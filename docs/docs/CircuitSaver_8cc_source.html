42 void CircuitSaver::loadCircuit(std::string name, Scene* s) {
43  std::string path = getPath(name);
44 
45  // Check if file exists, if not don't try to open it.
46  if(not FileUtils::pathExists(path)){
47  std::cerr << "Directory: '" << path << "' does not exist!" << std::endl;
48  return;
49  }
50 
51  std::cout << "Load: '" << path << "'" << std::endl;
52 
53  std::ifstream in(path.c_str());
54 
55  // Load the JSON string into a JSON object.
56 
57  std::stringstream rawData;
58  rawData << in.rdbuf();
59  in.close();
60 
61  json::jobject data = json::jobject::parse(rawData.str());
62 
63 
64  std::map<int, UIComponent*> components;
65 
66  std::vector<json::jobject> parts;
67  parts = data["parts"];
68 
69  for(int i=0 ; i<parts.size() ; i++){
70  json::jobject component = parts[i]["component"];
71  int uid = parts[i]["id"];
72  UIComponent* comp = NULL;
73 
74  int id = component["type"];
75 
76  // Create UIComponent for each json component.
77  switch(id) {
78  case UI_BATTERY: {
79  comp = (UIComponent*) new Battery((double) component["voltage"], (int) component["state"]);
80  break;
81  }
82  case UI_RESISTOR: {
83  comp = (UIComponent*) new Resistor((double) component["resistance"]);
84  break;
85  }
86  case UI_WIRE: {
87  comp = (UIComponent*) new Wire((double) component["length"], (double) component["area"], component["material"]);
88  break;
89  }
90  case UI_SWITCH: {
91  comp = (UIComponent*) new Switch((int) component["state"]);
92  break;
93  }
94 
95  default:
96  break;
97  }
98 
99  // If we actually made a component.
100  if(comp != NULL){
101  // Set the coordinates of the component
102  double x = (double) component["pos"].array(0);
103  double y = (double) component["pos"].array(1);
104 
105  // Validate x, y minimum value of 0.
106  x = x<0? 0 : x;
107  y = y<0? 0 : y;
108 
109  // Validate x, y maximum value of the size of the scene.
110  x = x>s->width? s->width : x;
111  y = y>s->height? s->height : y;
112 
113  comp->setPos(x, y);
114 
115  // Get the ID of the component.
116  comp->componentId = uid;
117 
118  // Set the max ID by comparing this ID to the current maximum.
119  if(comp->componentId > UIComponent::currentId){
120  UIComponent::currentId = comp->componentId;
121  }
122 
123  // Add the component to the scene.
124  s->addItem(comp);
125  // Add the component to the ID map.
126  components[comp->componentId] = comp;
127  }
128 
129  }
130 
131  parts = data["parts"];
132 
133  for(int i=0 ; i<parts.size() ; i++){
134  json::jobject part = parts[i];
135  // Set the start item for all the arrows.
136  UIComponent* startItem = components[(int)part["id"]];
137 
138 
139  std::vector<int> connections = part["connections"];
140  for(int x=0 ; x<connections.size() ; x++){
141  int conn = connections[x];
142  // For each connection in the JSON data get the component from the ID map.
143  UIComponent* endItem = components[conn];
144 
145  // Create a line between the start and end components.
146  Line* line = new Line(startItem, endItem);
147  startItem->addLine(line);
148  endItem->addLine(line);
149 
150  // Add the line to the scene and update it.
151  s->addItem(line);
152  line->update();
153  }
154  }

158 std::string CircuitSaver::serialiseCircuit(std::string name, SceneItems items) {
159  json::jobject out;
160  out["name"] = name;
161 
162  std::list<QGraphicsItem*> graphicsItems(items.components.begin(), items.components.end());
163 
164  AnalysisMapper am(graphicsItems);
165  // Use AnalysisMapper to simplify the code to generate a graph of the components.
166  Graph graph = am.makeGraph();
167 
168  std::vector<json::jobject> parts;
169 
170  foreach(UIComponent* uicomp, graph.keys()){
171  json::jobject sect;
172  // Turn the component into JSON
173  json::jobject comp = serialiseUIComponent(uicomp);
174  std::vector<int> connections;
175 
176  std::vector<UIComponent*> components = graph[uicomp];
177 
178  // For each connection add the ID of the connected component.
179  for(int i=0 ; i<components.size() ; i++){
180  connections.push_back(components[i]->componentId);
181  }
182 
183  // Add the data to the JSON map.
184  sect["component"] = comp;
185  sect["connections"] = connections;
186  sect["id"] = uicomp->componentId;
187 
188  parts.push_back(sect);
189  }
190 
191  out["parts"] = parts;
192 
193  return (std::string) out;
194 }

197 json::jobject CircuitSaver::serialiseUIComponent(UIComponent* comp) {
198  json::jobject out;
199 
200  // Get the type of component.
201  out["type"] = comp->getId();
202 
203  // Different components have different saved values.
204  switch(comp->getId()){
205  case UI_BATTERY:
206  // getVoltageValue as this gives the raw value, getVoltage changes the voltage by error
207  // checking and including the state value.
208  out["voltage"] = ((Battery*) comp)->getVoltageValue();
209  out["state"] = ((Battery*) comp)->getState();
210  break;
211  case UI_RESISTOR:
212  out["resistance"] = ((Resistor*) comp)->getResistance();
213  break;
214  case UI_WIRE:
215  out["area"] = ((Wire*) comp)->getArea();
216  out["length"] = ((Wire*) comp)->getLength();
217  out["material"] = ((Wire*) comp)->getMaterial();
218  break;
219  case UI_SWITCH:
220  out["state"] = ((Switch*) comp)->getState();
221  break;
222  }
223 
224  // Add the coordinates of the component in a list of [x,y].
225  std::vector<double> pos;
226  pos.push_back(comp->pos().x());
227  pos.push_back(comp->pos().y());
228  out["pos"] = pos;
229 
230  return out;
231 }
