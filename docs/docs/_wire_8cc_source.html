62double Wire::getResistance() {
63 // Get the length, convert cm to m by dividing by 100.
64 long double length = lengthSpinner->value()/100L;
65
66 // Get the area, convert mm^2 to m^2 by dividing by 1000^2
67 long double area = areaSpinner->value()/1e6L;
68
69 // Get the material by converting the wire combobox to text and then indexing the resistivities map.
70 std::string material = wireCombo->currentText().toStdString();
71 long double resistivity = resistivities.at(material);
72
73 // Use the resistivity formula p=(RA/l), rearranged to R=(pl)/A
74 long double resistance = (resistivity*length)/area;
75
76 return (double) resistance;
