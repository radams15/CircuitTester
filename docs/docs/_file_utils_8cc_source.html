48bool FileUtils::createDirTree(std::string tree){
49#if UNIX
50 // Adapted from https://stackoverflow.com/questions/675039/how-can-i-create-directory-tree-in-c-linux
51 mode_t mode = 0755;
52 if (mkdir(tree.c_str(), mode) == 0) { // Already exists.
53 return true;
54 }
55
56 switch (errno){
57 case ENOENT: // File does not exist.
58 {
59 int pos = tree.find_last_of('/');
60
61 // No separators in the path, all finished.
62 if (pos == std::string::npos){
63 return false;
64 }
65
66 // Recursively create again with path up to the last separator.
67 if (!createDirTree( tree.substr(0, pos) )){
68 return false;
69 }
70 }
71 // Check if file exists.
72 return 0 == mkdir(tree.c_str(), mode);
73
74 case EEXIST: // File already exists
75 return pathExists(tree);
76
77 default:
78 return false;
79 }
80#elif WINDOWS
81 // Adapted from https://stackoverflow.com/questions/1517685/recursive-createdirectory#12745750
82 size_t pos = 0;
83 do {
84 // Find position of the last path separator.
85 pos = tree.find_first_of("\\/", pos + 1);
86 // CreateDirectory repeatedly until pos is end of the string.
87 CreateDirectory(tree.substr(0, pos).c_str(), NULL);
88 } while (pos != std::string::npos);
89
90 return false;
91#else
92#error Cannot determine OS (createDirTree)!
93#endif
94}

151bool FileUtils::endsWith(std::string fullString, std::string ending) {
152 if (fullString.length() >= ending.length()) {
153 // Compare the last n (length of ending) chars of fullString to all the chars of ending.
154 // Compare returns 0 if they are identical.
155 return (fullString.compare(fullString.length() - ending.length(), ending.length(), ending) == 0);
156 }
157

161std::string FileUtils::replace(std::string subj, std::string from, std::string to) {
162 // Find position of the from string in the subj string
163 size_t pos = subj.find(from);
164 // If the position is not a position, return the unmodified string.
165 if(pos == std::string::npos) {
166 return subj;
167 }
168
169 // Replace the position pos in subject with to.
170 subj.replace(pos, from.length(), to);
171 return subj;
172}

175std::vector<std::string> FileUtils::getSaveFiles(){
176 std::vector<std::string> out;
177#if UNIX
178 DIR *d; // https://stackoverflow.com/questions/4204666/how-to-list-files-in-a-directory-in-a-c-program
179 struct dirent *dir;
180 d = opendir(getSaveDir().c_str());
181 if (d) {
182 while ((dir = readdir(d)) != NULL) {
183 if(endsWith(dir->d_name, ".cir")){
184 out.push_back(replace(std::string(dir->d_name), ".cir", ""));
185 }
186 }
187 closedir(d);
188 }
189#elif WINDOWS
190 WIN32_FIND_DATA fdFile; // https://stackoverflow.com/questions/2314542/listing-directory-contents-using-c-and-windows
191 HANDLE hFind = NULL;
192
193 char sPath[2048];
194
195 sprintf(sPath, "%s\\*.*", getSaveDir().c_str());
196
197 if((hFind = FindFirstFile(sPath, &fdFile)) == INVALID_HANDLE_VALUE){
198 printf("Path not found: [%s]\n", getSaveDir().c_str());
199 return out;
200 }
201
202 do{
203 if(strcmp(fdFile.cFileName, ".") != 0 && strcmp(fdFile.cFileName, "..") != 0){
204 sprintf(sPath, "%s\\%s", getSaveDir().c_str(), fdFile.cFileName);
205
206 if(fdFile.dwFileAttributes &FILE_ATTRIBUTE_DIRECTORY){}else{
207 if(endsWith(sPath, ".cir")){
208 out.push_back(replace(std::string(fdFile.cFileName), ".cir", ""));
209 }
210 }
211 }
212 } while(FindNextFile(hFind, &fdFile));
213
214 FindClose(hFind);
215#else
216#error Cannot determine OS (getSaveFiles)!
217#endif
218 return out;
219}
