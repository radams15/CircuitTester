51void SettingsMenu::startAnimation(bool checked) {
52 // Change arrow depending on direction. E.g. right arrow to close if open, left arrow to open if closed.
53 toggleButton->setArrowType(checked? Qt::RightArrow : Qt::LeftArrow);
54
55 // Set the animation to move forward or backward, e.g. to open move the animation forward,
56 // to close move the animation backward.
57 toggleAnimation->setDirection(checked? QAbstractAnimation::Forward : QAbstractAnimation::Backward);
58 // Begin the animation.
59 toggleAnimation->start();
60}

62void SettingsMenu::setContentLayout(QLayout* contentLayout) {
63 contentArea->setLayout(contentLayout);
64
65 // Width of the content area when it should be collapsed.
66 int collapsedWidth = sizeHint().width() - contentArea->maximumWidth();
67 int contentWidth = 200;
68
69 // Run through each animation except the last to set the properties.
70 for(int i=0 ; i<toggleAnimation->animationCount()-1 ; i++){
71 auto* HelpMenuAnimation = (QPropertyAnimation*) toggleAnimation->animationAt(i);
72 HelpMenuAnimation->setDuration(animationDuration);
73 HelpMenuAnimation->setStartValue(collapsedWidth);
74 HelpMenuAnimation->setEndValue(collapsedWidth + contentWidth);
75 }
76
77 auto* contentAnimation = (QPropertyAnimation*) toggleAnimation->animationAt(toggleAnimation->animationCount()-1);
78 contentAnimation->setDuration(animationDuration);
79 contentAnimation->setStartValue(0);
80 contentAnimation->setEndValue(contentWidth);
