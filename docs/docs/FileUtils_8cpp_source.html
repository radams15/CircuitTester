47 bool FileUtils::createDirTree(std::string tree){
48 #if UNIX
49  // Adapted from https://stackoverflow.com/questions/675039/how-can-i-create-directory-tree-in-c-linux
50  mode_t mode = 0755;
51  if (mkdir(tree.c_str(), mode) == 0) { // Already exists.
52  return true;
53  }
54 
55  switch (errno){
56  case ENOENT: // File does not exist.
57  {
58  int pos = tree.find_last_of('/');
59 
60  // No separators in the path, all finished.
61  if (pos == std::string::npos){
62  return false;
63  }
64 
65  // Recursively create again with path up to the last separator.
66  if (!createDirTree( tree.substr(0, pos) )){
67  return false;
68  }
69  }
70  // Check if file exists.
71  return 0 == mkdir(tree.c_str(), mode);
72 
73  case EEXIST: // File already exists
74  return pathExists(tree);
75 
76  default:
77  return false;
78  }
79 #elif WINDOWS
80  // Adapted from https://stackoverflow.com/questions/1517685/recursive-createdirectory#12745750
81  size_t pos = 0;
82  do {
83  // Find position of the last path separator.
84  pos = tree.find_first_of("\\/", pos + 1);
85  // CreateDirectory repeatedly until pos is end of the string.
86  CreateDirectory(tree.substr(0, pos).c_str(), NULL);
87  } while (pos != std::string::npos);
88 
89  return false;
90 #else
91 #error Cannot determine OS (createDirTree)!
92 #endif
93 }
