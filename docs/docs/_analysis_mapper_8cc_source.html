45std::map<UIComponent*, ComponentValue> AnalysisMapper::getSolution() {
46 Graph graph = makeGraph();
47
48 printGraph(graph);
49
50 std::vector<Component> mNAComponents;
51
52 // Map of UIComponent:Component to help to return the correct values to the correct UIComponent.
53 std::map<UIComponent*, Component> mNAMap;
54
55 for(auto node : graph){
56 Component* component;
57
58 // Get the type of the node (node.first is node, node.second is connections).
59 switch(node.first->getId()){
60 case UI_BATTERY:
61 component = new Component(node.first->n0, node.first->n1, MNA_BATTERY, ((Battery*)node.first)->getVoltage());
62 break;
63
64 // Resistors, wires and switches all have resistances.
65 case UI_RESISTOR: case UI_WIRE: case UI_SWITCH: case UI_AMMETER: case UI_VOLTMETER:
66 component = new Component(node.first->n0, node.first->n1, MNA_RESISTOR, ((ResistiveElement*)node.first)->getResistance());
67 break;
68
69 default:
70 continue;
71 }
72
73 // Add the new component to the list of components.
74 mNAComponents.push_back(*component);
75
76 // Add the component to the list to set the (UIComponent => Component).
77 mNAMap.insert(std::make_pair(node.first, *component));
78 }
79
80 Circuit cir(mNAComponents);
81
82 auto sol = cir.solve();
83
84 std::map<UIComponent*, ComponentValue> out;
85
86 for(auto it : mNAMap){
87 switch(it.second.type){
88 // Resistors have a current and a voltage.
89 case MNA_RESISTOR:
90 // Add the voltage and the current of the Component into the solution map
91 // with the key of the UIComponent.
92 out[it.first] = {
93 sol.getVoltage(it.second),
94 sol.getCurrent(it.second)
95 };
96 break;
97
98 // Batteries have only a voltage, not a current.
99 case MNA_BATTERY:
100 // Add the voltage of the Component into the solution map.
101 // with the key of the UIComponent.
102 out[it.first] = {
103 sol.getVoltage(it.second),
104 NAN
105 };
106 break;
107
108 default:
109 break;
110 }
111
112 switch(it.first->getId()){
113 case UI_VOLTMETER:
114 out[it.first] = {
115 sol.getVoltage(it.second),
116 NAN
117 };
118 break;
119
120 case UI_AMMETER:
121 out[it.first] = {
122 NAN,
123 sol.getCurrent(it.second)
124 };
125 break;
126
127 default:
128 break;
129 }
130 }
131
132 return out;
133}

135Path* AnalysisMapper::findShortestPath(Graph *graph, UIComponent *start, UIComponent *end) {
136 std::vector<UIComponent*> explored;
137
138 std::queue<std::vector<UIComponent*>*> q;
139
140 auto v = new std::vector<UIComponent*>;
141 v->push_back(start);
142 q.emplace(v);
143
144 // Is the path to the same node?
145 // If so just return the end node.
146 if(start == end){
147 auto out = new Path;
148 out->push_back(end);
149 return out;
150 }
151
152 while(! q.empty()){
153 Path* path = q.front();
154 q.pop();
155
156 // Get end node from the path.
157 UIComponent* node = path->at(path->size()-1);
158
159 // If node has not already been explored
160 if(std::find(explored.begin(), explored.end(), node) == explored.end()){
161 auto neighbors = graph->at(node);
162
163 for(auto neighbor : neighbors){
164 Path* new_path = new Path;
165 std::copy(path->begin(), path->end(), back_inserter(*new_path));
166 new_path->push_back(neighbor);
167 q.emplace(new_path);
168
169 // Have we reached the end node?
170 // If so return the path.
171 if(neighbor == end){
172 return new_path;
173 }
174 }
175
176 explored.push_back(node);
177 }
178 }
179
180 // If no path was found return an empty path.
181 return new Path;
182}

184Graph AnalysisMapper::makeGraph() {
185 // This is a map of UIComponent:list of UIComponents to show a matrix as an adjacency list.
186 Graph graph;
187
188 for(auto comp : components){
189 std::vector<UIComponent*> connections;
190
191 for(auto a : comp->lines){
192 // If the arrow goes to the component rather than away from it, ignore it.
193 if(comp == a->endItem()){
194 continue;
195 }
196 // Add a connection to the list.
197 connections.push_back((UIComponent*) a->endItem());
198 }
199
200 // Add the list to the graph.
201 graph[comp] = connections;
202 }
203
204 // TODO select correct start node
205 auto start_node = components[0];
206
207 for(auto n : graph){
208 // For each node set the node 0 to the distance from the start node.
209 auto* path = findShortestPath(&graph, start_node, n.first);
210
211 n.first->n0 = path->size()-1;
212
213 // Temporarily set node 1 to 0.
214 n.first->n1 = 0;
215 }
216
217 for(auto node : graph){
218 node.first->connections.clear();
219
220 for(auto connectedComp : graph.at(node.first)){
221 // For each connection to this node, set node 1 to the node 0 of the node it is connected to.
222 node.first->n1 = connectedComp->n0;
223 }
224 }
225
226 return graph;
227}
