343 std::string dtos(double in){
344  /* Alloc 10 bytes: will take up 4 bytes for the max integer part (1000 max), 2 for the 2dp,
345  1 for the point char, and the null byte.*/
346  auto outa = (char*) calloc(8, sizeof(char));
347 
348  // Write the double to 2 decimal places to the outa char array.
349  sprintf(outa, "%.2f", in);
350 
351  // Turn the char array to a std::string.
352  auto outs = std::string(outa);
353 
354  // Free the memory as we copied it to an std::string and no longer need it.
355  free((void*) outa);
356 
357  return outs;
358 }

360 void MainWindow::runSimulation() {
361  // Is the scene empty? If it is, don't run as this crashes the program.
362  if(scene->items().empty()){
363  return;
364  }
365 
366  // Initialise AnalysisMapper with a std::list of scene items.
367  AnalysisMapper mapper(toStdList(scene->items()));
368 
369  auto sol = mapper.getSolution();
370 
371  // Remove all existing text boxes.
372  scene->removeAllText();
373 
374  // 'it' is an iterator of a map of UIComponent: ComponentValue.
375  for(auto it : sol){
376  std::stringstream ss;
377 
378  if (std::to_string(it.second.voltage) != "nan"){ //TODO FIX THIS MONSTROSITY
379  ss << "Voltage: ";
380  if(it.second.voltage < 1000){
381  ss << dtos(it.second.voltage);
382  }else if(it.second.voltage <= 0.01) {
383  ss << "0";
384  }else{
385  ss << "\u221E"; // Infinity symbol unicode escape.
386  }
387  ss << "V";
388  }
389 
390  if (std::to_string(it.second.current) != "nan"){ //TODO FIX THIS MONSTROSITY
391  ss << "\nCurrent: ";
392  if(it.second.current < 1000){
393  ss << dtos(it.second.current);
394  }else if(it.second.current < 0.01) {
395  ss << "0";
396  }else{
397  ss << "\u221E"; // Infinity symbol unicode escape.
398  }
399  ss << "A";
400  }
401 
402  // Create a textbox.
403  auto textBox = new SceneText(ss.str());
404  auto componentPos = it.first->pos();
405  scene->addItem(textBox);
406  // Add textbox underneath the component.
407  textBox->setPos(QPointF(componentPos.x()+100, componentPos.y()+200));
408  }

421 void MainWindow::saveScene() {
422  std::vector<UIComponent*> components;
423  std::vector<Line*> lines;
424 
425  // Sort the items into components and lines.
426  for(QGraphicsItem *i : scene->items()){
427  if(IS_TYPE(UIComponent, i)) {
428  // If it is a component, cast and add to components.
429  components.push_back((UIComponent*) i);
430  } else if(IS_TYPE(Line, i)) {
431  // If it is a line, cast and add to lines.
432  lines.push_back((Line*) i);
433  }
434  }
435 
436  // Is there already a circuit open? If not name this circuit.
437  if(currentOpenedCircuit.empty()){
438  // Prompt the user for the name of the circuit to save.
439  std::string name = QInputDialog::getText(this, tr("Circuit Name"), tr("Name")).toStdString();
440  currentOpenedCircuit = name;
441  }
442 
443  // If the user still didn't name the circuit then quit the save.
444  if(currentOpenedCircuit.empty()){
445  return;
446  }
447 
448  CircuitSaver::saveCircuit(currentOpenedCircuit, SceneItems{components, lines});
449 
450  // Indicate to the user the name of the open file.
451  setWindowTitle(tr(("Circuit Simulator: " + currentOpenedCircuit).c_str()));
452 }
