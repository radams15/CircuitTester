40void CircuitSaver::loadCircuit(std::string name, Scene* s) {
41 std::string path = getPath(name);
42
43 // Check if file exists, if not don't try to open it.
44 if(not FileUtils::pathExists(path)){
45 std::cerr << "Directory: '" << path << "' does not exist!" << std::endl;
46 return;
47 }
48
49 std::cout << "Load: '" << path << "'" << std::endl;
50
51 std::ifstream in(path);
52
53 // Load the JSON string into a JSON object.
54 json data;
55 in >> data;
56 in.close();
57
58 std::map<int, UIComponent*> components;
59
60 for(auto part : data["parts"]){
61 UIComponent* comp = nullptr;
62
63 // Create UIComponent for each json component.
64 switch(part["component"]["type"].get<int>()) {
65 case UI_BATTERY: {
66 comp = new Battery(part["component"]["voltage"].get<double>(), part["component"]["state"].get<bool>());
67 break;
68 }
69 case UI_RESISTOR: {
70 comp = new Resistor(part["component"]["resistance"].get<double>());
71 break;
72 }
73 case UI_WIRE: {
74 comp = new Wire(part["component"]["length"].get<double>(), part["component"]["area"].get<double>(), part["component"]["material"].get<std::string>());
75 break;
76 }
77 case UI_SWITCH: {
78 comp = new Switch(part["component"]["state"].get<bool>());
79 break;
80 }
81
82 default:
83 break;
84 }
85
86 // If we actually made a component.
87 if(comp != nullptr){
88 // Set the coordinates of the component
89 auto x = part["component"]["pos"][0].get<double>();
90 auto y = part["component"]["pos"][1].get<double>();
91
92 auto sceneSize = s->sceneRect();
93
94 // Validate x, y minimum value of 0.
95 x = x<0? 0 : x;
96 y = y<0? 0 : y;
97
98 // Validate x, y maximum value of the size of the scene.
99 x = x>s->width? s->width : x;
100 y = y>s->height? s->height : y;
101
102 comp->setPos(x, y);
103
104 // Get the ID of the component.
105 comp->componentId = part["id"].get<int>();
106
107 // Set the max ID by comparing this ID to the current maximum.
108 if(comp->componentId > UIComponent::currentId){
109 UIComponent::currentId = comp->componentId;
110 }
111
112 // Add the component to the scene.
113 s->addItem(comp);
114 // Add the component to the ID map.
115 components[comp->componentId] = comp;
116 }
117 }
118
119 for(auto part : data["parts"]){
120 // Set the start item for all the arrows.
121 auto startItem = components[part["id"].get<int>()];
122
123 for(auto conn : part["connections"]){
124 // For each connection in the JSON data get the component from the ID map.
125 auto endItem = components[conn.get<int>()];
126
127 // Create a line between the start and end components.
128 auto* line = new Line(startItem, endItem);
129 startItem->addLine(line);
130 endItem->addLine(line);
131
132 // Add the line to the scene and update it.
133 s->addItem(line);
134 line->update();
135 }
136 }
137
138}

140std::string CircuitSaver::serialiseCircuit(std::string name, SceneItems items) {
141 json out = json::object();
142 out["name"] = name;
143
144 std::list<QGraphicsItem*> graphicsItems;
145
146 // Cast each UIComponent into a QGraphicsItem for AnalysisMapper.
147 std::transform(items.components.begin(), items.components.end(), std::back_inserter(graphicsItems), [](UIComponent* c){
148 return (QGraphicsItem*) c;
149 });
150
151 AnalysisMapper am(graphicsItems);
152 // Use AnalysisMapper to simplify the code to generate a graph of the components.
153 Graph graph = am.makeGraph();
154
155 json parts = json::array();
156
157 for(auto it : graph){
158 json sect = json::object();
159 // Turn the component into JSON
160 json comp = serialiseUIComponent(it.first);
161 json connections = json::array();
162
163 // For each connection add the ID of the connected component.
164 for(auto c : it.second){
165 connections.push_back(c->componentId);
166 }
167
168 // Add the data to the JSON map.
169 sect["component"] = comp;
170 sect["connections"] = connections;
171 sect["id"] = it.first->componentId;
172
173 parts.push_back(sect);
174 }
175
176 out["parts"] = parts;
177
178 if(items.scene != nullptr){
179 // Only create the image if we have a scene to convert.
180 out["image"] = sceneToImage(items.scene);
181 }
182
183 return out.dump(4);
184}

186std::string CircuitSaver::sceneToImage(Scene *s, QImage::Format format) {
187 // Create an image the size of the scene.
188 auto image = QImage(s->itemsBoundingRect().size().toSize(), format);
189 // Make the image white.
190 image.fill(QColor(Qt::color0).rgb());
191
192 QPainter painter;
193
194 // Paint onto image.
195 painter.begin(&image);
196 painter.setRenderHint(QPainter::Antialiasing);
197
198 // Render the scene onto the image.
199 s->render(&painter, image.rect(), s->itemsBoundingRect());
200 painter.end();
201
202 QByteArray ba;
203 QBuffer buf(&ba);
204 buf.open(QIODevice::WriteOnly);
205 // Save the image onto the buffer, which writes the data to the QByteArray.
206 image.save(&buf, "PNG");
207
208 // Convert the QByteArray to a base64 string and return.
209 return ba.toBase64().toStdString();
210}

212json CircuitSaver::serialiseUIComponent(UIComponent* comp) {
213 json out = json::object();
214
215 // Get the type of component.
216 out["type"] = comp->getId();
217
218 // Different components have different saved values.
219 switch(comp->getId()){
220 case UI_BATTERY:
221 // getVoltageValue as this gives the raw value, getVoltage changes the voltage by error
222 // checking and including the state value.
223 out["voltage"] = ((Battery*) comp)->getVoltageValue();
224 out["state"] = ((Battery*) comp)->getState();
225 break;
226 case UI_RESISTOR:
227 out["resistance"] = ((Resistor*) comp)->getResistance();
228 break;
229 case UI_WIRE:
230 out["area"] = ((Wire*) comp)->getArea();
231 out["length"] = ((Wire*) comp)->getLength();
232 out["material"] = ((Wire*) comp)->getMaterial();
233 break;
234 case UI_SWITCH:
235 out["state"] = ((Switch*) comp)->getState();
236 break;
237 }
238
239 // Add the coordinates of the component in a list of [x,y].
240 out["pos"] = json::array({comp->pos().x(), comp->pos().y()});
241
242 return out;
243}

262void CircuitSaver::importCircuit(std::string path) {
263 // Regex removes path and extracts file name, e.g. "/home/rhys/test.cir", the output is an array ["/home/rhys/test.cir", "test"]
264 std::regex nameRegex(R"(.*(?:\/|\\)(.*).cir)");
265 std::cmatch match;
266
267 // Search the string to match the regex above.
268 if(! std::regex_search(path.c_str(), match, nameRegex)){
269 std::cerr << "Cannot regex match: " << path << std::endl;
270 return;
271 }
272
273 // Match[1] is the 1st match group which is the file name.
274 if(match[1] != ""){
275 auto saveFile = CircuitSaver::getPath(match[1]);
276 FileUtils::copyFile(path, saveFile);
277 }
