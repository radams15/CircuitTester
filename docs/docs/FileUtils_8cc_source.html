48 bool FileUtils::createDirTree(std::string tree){
49 #if UNIX
50  // Adapted from https://stackoverflow.com/questions/675039/how-can-i-create-directory-tree-in-c-linux
51  mode_t mode = 0755;
52  if (mkdir(tree.c_str(), mode) == 0) { // Already exists.
53  return true;
54  }
55 
56  switch (errno){
57  case ENOENT: // File does not exist.
58  {
59  int pos = tree.find_last_of('/');
60 
61  // No separators in the path, all finished.
62  if (pos == std::string::npos){
63  return false;
64  }
65 
66  // Recursively create again with path up to the last separator.
67  if (!createDirTree( tree.substr(0, pos) )){
68  return false;
69  }
70  }
71  // Check if file exists.
72  return 0 == mkdir(tree.c_str(), mode);
73 
74  case EEXIST: // File already exists
75  return pathExists(tree);
76 
77  default:
78  return false;
79  }
80 #elif WINDOWS
81  // Adapted from https://stackoverflow.com/questions/1517685/recursive-createdirectory#12745750
82  size_t pos = 0;
83  do {
84  // Find position of the last path separator.
85  pos = tree.find_first_of("\\/", pos + 1);
86  // CreateDirectory repeatedly until pos is end of the string.
87  CreateDirectory(tree.substr(0, pos).c_str(), NULL);
88  } while (pos != std::string::npos);

90  return false;

92 #error Cannot determine OS (createDirTree)!
93 #endif
