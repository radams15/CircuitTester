37  for(auto node : graph){

40  // Get the type of the node (node.first is node, node.second is connections).
41  switch(node.first->getId()){

43  component = new Component(node.first->n0, node.first->n1, MNA_BATTERY, ((Battery*)node.first)->getVoltage());

45

48  component = new Component(node.first->n0, node.first->n1, MNA_RESISTOR, ((ResistiveElement*)node.first)->getResistance());
49  break;
50

53  }
54

56  mNAComponents->push_back(component);
57 
58  // Add the component to the list to set the (UIComponent => Component).
59  mNAMap->insert(std::make_pair(node.first, component));

117 Path* AnalysisMapper::findShortestPath(Graph *graph, UIComponent *start, UIComponent *end) {
118  std::vector<UIComponent*> explored;
119 
120  std::queue<std::vector<UIComponent*>*> q;
121 
122  auto v = new std::vector<UIComponent*>;
123  v->push_back(start);
124  q.emplace(v);
125 
126  // Is the path to the same node?
127  // If so just return the end node.
128  if(start == end){
129  auto out = new Path;
130  out->push_back(end);
131  return out;
132  }
133 
134  while(! q.empty()){
135  Path* path = q.front();
136  q.pop();
137 
138  // Get end node from the path.
139  UIComponent* node = path->at(path->size()-1);
140 
141  // If node has not already been explored
142  if(std::find(explored.begin(), explored.end(), node) == explored.end()){
143  auto neighbors = graph->at(node);
144 
145  for(auto neighbor : neighbors){
146  Path* new_path = new Path;
147  std::copy(path->begin(), path->end(), back_inserter(*new_path));
148  new_path->push_back(neighbor);
149  q.emplace(new_path);
150 
151  // Have we reached the end node?
152  // If so return the path.
153  if(neighbor == end){
154  return new_path;
155  }
156  }
157 
158  explored.push_back(node);
159  }
160  }
161 
162  // If no path was found return an empty path.
163  return new Path;
164 }

166 Graph AnalysisMapper::makeGraph() {
167  // This is a map of UIComponent:list of UIComponents to show a matrix as an adjacency list.
168  Graph graph;
169 
170  for(auto comp : components){
171  std::vector<UIComponent*> connections;
172 
173  for(auto a : comp->lines){
174  // If the arrow goes to the component rather than away from it, ignore it.
175  if(comp == a->endItem()){
176  continue;
177  }
178  // Add a connection to the list.
179  connections.push_back((UIComponent*) a->endItem());
180  }
181 
182  // Add the list to the graph.
183  graph[comp] = connections;
184  }
185 
186  // TODO select correct start node
187  auto start_node = components[0];
188 
189  for(auto n : graph){
190  // For each node set the node 0 to the distance from the start node.
191  auto* path = findShortestPath(&graph, start_node, n.first);
192 
193  n.first->n0 = path->size()-1;
194 
195  // Temporarily set node 1 to 0.
196  n.first->n1 = 0;
197  }
198 
199  for(auto node : graph){
200  node.first->connections.clear();
201 
202  for(auto connectedComp : graph.at(node.first)){
203  // For each connection to this node, set node 1 to the node 0 of the node it is connected to.
204  node.first->n1 = connectedComp->n0;
205  }
206  }
207 
208  return graph;
209 }
