16  T out = vec.back(); // get last element

104 std::vector<Term *>* MNACircuit::getCurrents(int node, int side) {
105  /*Convert the side to the current direction - if side is 0, current direction
106  * is 1, otherwise the current direction is -1*/
107  int currentDirection = side == 0 ? 1 : -1;
108 
109  // Check that the side is valid - either 0 or 1.
110  if(side != 0 and side != 1) throw std::invalid_argument("Invalid Side!");
111 
112  // The current terms to return.
113  auto* out = new std::vector<Term*>;
114 
115  for(auto b : batteries){
116  // Convert node index into node value.
117  int n = side == 0? b->n0 : b->n1;
118 
119  // If this is the node that was specified in the parameter.
120  if(n == node) {
121  // Batteries have an unknown current.
122  out->push_back(new Term(currentDirection, new UnknownCurrent(b)));
123  }
124  }
125 
126  for(auto r : resistors) {
127  // Convert node index into node value.
128  int n = side == 0 ? r->n0 : r->n1;
129 
130  // If correct node and a zero-resistance resistor.
131  if (n == node && r->value == 0) {
132  // The resistor has an unknown current, make the object.
133  out->push_back(new Term(currentDirection, new UnknownCurrent(r)));
134  }else if(n == node && r->value != 0){
135  // If the resistance is not 0, the resistor has an unknown voltage.
136  // The voltage enters and leaves the component, so there are 2 terms.
137  out->push_back(new Term(-currentDirection / r->value, new UnknownVoltage(r->n1 )));
138  out->push_back(new Term(currentDirection / r->value, new UnknownVoltage(r->n0 )));
139  }
140  }
141 
142  return out;
143 }

145 std::vector<int>* MNACircuit::getRefNodes() {
146  auto* out = new std::vector<int>;
147 
148  // For every component,
149 
150  std::vector<int> toVisit = nodes;
151 
152  while(! toVisit.empty()){
153  // Get the element to visit next from toVisit.
154  int refNodeId = toVisit.at(0);
155 
156  // Add the reference node to the output list
157  out->push_back(refNodeId);
158 
159  std::vector<int>* connectedNodes = getConnectedNodes(refNodeId);
160 
161  // Remove every node that is directly connected to this reference node,
162  // as there cannot be 2 nodes that are connected but are both reference nodes.
163  for(auto c : *connectedNodes){
164  // Remove c from toVisit by iterating over every item and removing
165  // the matching one.
166  toVisit.erase(std::remove_if(toVisit.begin(), toVisit.end(), [c](int i){
167  return i == c;
168  }));
169  }
170  }
171 
172  return out;
173 }

175 std::vector<int>* MNACircuit::getConnectedNodes(int node) {
176  // A queue of visited nodes from this node.
177  auto* visited = new std::vector<int>;
178 
179  // A list of nodes that we want to visit in the future.
180  std::vector<int> toVisit = {node};
181 
182  // This is basically a breadth first iteration of every element connected to this node.
183  while(not toVisit.empty()){
184  // Get the top vector item.
185  int nodeToVisit = vecPopFront<int>(toVisit);
186 
187  visited->push_back(nodeToVisit);
188 
189  for(auto e : elements){
190  if(e->contains(nodeToVisit)){
191  // Get the node leaving the element.
192  int oppositeNode = e->opposite(nodeToVisit);
193 
194  // Visit the opposite node in the future if we have not already visited it.
195  if(! std::count(visited->begin(), visited->end(), oppositeNode)){
196  toVisit.push_back(oppositeNode);
197  }
198  }
199  }
200  }
201 
202  // Remove duplicate values, just in case there was an error.
203  visited->erase(unique( visited->begin(), visited->end() ), visited->end());
204  return visited;
205 }

207 std::vector<Equation *>* MNACircuit::getEquations() {
208  auto* equations = new std::vector<Equation *>;
209 
210  std::vector<int>* refNodeIds = getRefNodes();
211 
212  for(auto r : *refNodeIds){
213  // Reference nodes have a voltage of zero as the voltage is relative to this node.
214  equations->push_back(new Equation(0, {
215  new Term(1, new UnknownVoltage(r))
216  }));
217  }
218 
219  for(auto n : nodes){
220  //Create an equation containing the total current
221  // source multiplied by each of the outgoing and incoming
222  // current nodes from the node.
223 
224  // If n not in refNodeIds.
225  if(not std::count(refNodeIds->begin(), refNodeIds->end(), n)){
226  // All current nodes entering node n at n1.
227  std::vector<Term *>* incoming = getCurrents(n, CUR_IN);
228  // All current nodes leaving node n at n0.
229  std::vector<Term *>* outgoing = getCurrents(n, CUR_OUT);
230 
231  //Currents leave at n0 and enter at n1 because conventional
232  // currents are opposite to actual charge flow.
233 
234  // Add together the two lists of terms.
235  auto* conserved = new std::vector<Term *>;
236  conserved->insert(conserved->end(), incoming->begin(), incoming->end());
237  conserved->insert(conserved->end(), outgoing->begin(), outgoing->end());
238 
239  // Create an equation of total current equals the sum of all the terms.
240  equations->push_back(new Equation(getCurrentTotal(n), *conserved));
241  }
242  }
243 
244  for(auto b : batteries){
245  // Make an equation for every battery.
246  equations->push_back(new Equation(b->value, {
247  // N0 is negative as voltage is lost.
248  // N1 is the positive as voltage is gained.
249  new Term(-1, new UnknownVoltage(b->n0)),
250  new Term(1, new UnknownVoltage(b->n1))
251  }));
252  }
253 
254  for(auto r : resistors){
255  if(r->value == 0){
256  //If resistance is 0, n0 and n1 are the same, as there
257  // may as well not be a component there
258  equations->push_back(new Equation(r->value, {
259  new Term(1, new UnknownVoltage(r->n0)),
260  new Term(-1, new UnknownVoltage(r->n1))
261  }));
262  }
263  }
264 
265  return equations;
266 }

286 MNASolution *MNACircuit::solve() {
287  std::vector<Equation *>* equations = getEquations();
288  std::vector<UnknownCurrent*>* unknownCurrents = getUnknownCurrents();
289  auto* unknownVoltages = new std::vector<UnknownVoltage*>;
290 
291  // Create an UnknownVoltage for each node as we don't know any voltage for any node.
292  for(auto v : nodes){
293  unknownVoltages->push_back(new UnknownVoltage(v));
294  }
295 
296  auto* unknowns = new std::vector<Unknown*>(unknownCurrents->begin(), unknownCurrents->end());
297 
298  unknowns->insert(unknowns->end(), unknownVoltages->begin(), unknownVoltages->end());
299 
300  //Make two matrices, one which is the width of the number of equations by the number
301  // of variables, and the other has a width of the number of equations but is only 1 wide
302  auto A = Eigen::MatrixXd(equations->size(), getNumVars()).setZero();
303  auto z = Eigen::MatrixXd(equations->size(), 1).setZero();
304 
305  for(int i=0 ; i<equations->size() ; i++){
306  //Apply the correct numbers of every equation onto the matrices,
307  // passing in a lambda to find the index of each
308  // term of the equation in class attribute unknowns.
309  equations->at(i)->apply(i, &A, &z, [this, unknowns](Unknown *u) {
310  return getElementIndex<Unknown>(unknowns, u);
311  });
312  }
313 
314  // Solve the matrix equation a = z
315  Eigen::MatrixXd x = A.fullPivLu().solve(z); // https://eigen.tuxfamily.org/dox/group__TutorialLinearAlgebra.html
316 
317  // A map of which nodes have which voltages on the circuit.
318  auto* voltageMap = new std::map<int, double>;
319 
320  for(auto v : *unknownVoltages){
321  // Get the voltage out from the solved matrix at the index of the element
322  // in the vector of unknowns. As the solved matrix stores
323  auto voltage = x(getElementIndex<Unknown>(unknowns, v));
324 
325  // Add to the voltage map
326  voltageMap->insert(std::pair<int, double>(v->node, voltage));
327  }
328 
329  //
330  std::vector<MNAElement*> elems;
331 
332  for(auto c : *unknownCurrents){
333  // Set the new current for each element, as the matrix column 0 has the solved currents.
334  c->element->currentSolution = x(getElementIndex<Unknown>(unknowns, c), 0);
335  elems.push_back(c->element);
336 
337  }
338 
339  return new MNASolution(*voltageMap, elems);
340 }
