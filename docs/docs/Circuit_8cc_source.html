14 T Circuit::vecPopFront(std::vector<T>& vec){
15  std::reverse(vec.begin(),vec.end()); // first becomes last, reverses the vector
16  T out = vec.back(); // get last component
17  vec.pop_back(); // pop last component off the vector
18  std::reverse(vec.begin(),vec.end()); // reverses it again, so the components are in the same order as before
19 
20  return out;

69 std::vector<Term> Circuit::getCurrents(int node, int side) {
70  /*Convert the side to the current direction - if side is 0, current direction
71  * is 1, otherwise the current direction is -1*/
72  int currentDirection = side == 0 ? 1 : -1;
73 
74  // Check that the side is valid - either 0 or 1.
75  if(side != 0 and side != 1) throw std::invalid_argument("Invalid Side!");
76 
77  // The current terms to return.
78  std::vector<Term> out;
79 
80  for(auto b : batteries){
81  // Convert node index into node value.
82  int n = side == 0? b.n0 : b.n1;
83 
84  // If this is the node that was specified in the parameter.
85  if(n == node) {
86  // Batteries have an unknown current.
87  out.push_back(Term(currentDirection, new UnknownCurrent(b)));
88  }
89  }
90 
91  for(auto r : resistors) {
92  // Convert node index into node value.
93  int n = side == 0 ? r.n0 : r.n1;
94 
95  // If correct node
96  if(n == node){
97  // The voltage enters and leaves the component, so there are 2 terms. This uses V=IR to find current, e.g. (1/R)*V = V/R = I
98  out.push_back(Term(-currentDirection / r.value, new UnknownVoltage(r.n1 )));
99  out.push_back(Term(currentDirection / r.value, new UnknownVoltage(r.n0 )));
100  }
101  }
102 
103  return out;
104 }

106 std::vector<int> Circuit::getRefNodes() {
107  std::vector<int> out;
108 
109  std::vector<int> toVisit = nodes;
110 
111  while(! toVisit.empty()){
112  // Get the component to visit next from toVisit.
113  int refNodeId = toVisit.at(0);
114 
115  // Add the reference node to the output list
116  out.push_back(refNodeId);
117 
118  auto connectedNodes = getConnectedNodes(refNodeId);
119 
120  // Remove every node that is directly connected to this reference node,
121  // as there cannot be 2 nodes that are connected but are both reference nodes.
122  for(auto c : connectedNodes){
123  // Remove c from toVisit by iterating over every item and removing
124  // the matching one.
125  toVisit.erase(std::remove_if(toVisit.begin(), toVisit.end(), [c](int i){
126  return i == c;
127  }), toVisit.end());
128  }
129  }
130 
131  return out;
132 }

134 std::vector<int> Circuit::getConnectedNodes(int node) {
135  // A queue of visited nodes from this node.
136  std::vector<int> visited;
137 
138  // A list of nodes that we want to visit in the future.
139  std::vector<int> toVisit = {node};
140 
141  // This is basically a breadth first iteration of every component connected to this node.
142  while(not toVisit.empty()){
143  // Get the top vector item.
144  int nodeToVisit = vecPopFront<int>(toVisit);
145 
146  visited.push_back(nodeToVisit);
147 
148  for(auto e : components){
149  if(e.contains(nodeToVisit)){
150  // Get the node leaving the component.
151  int oppositeNode = e.opposite(nodeToVisit);
152 
153  // Visit the opposite node in the future if we have not already visited it.
154  if(! std::count(visited.begin(), visited.end(), oppositeNode)){
155  toVisit.push_back(oppositeNode);
156  }
157  }
158  }
159  }
160 
161  // Remove duplicate values, just in case there was an error.
162  visited.erase(unique( visited.begin(), visited.end() ), visited.end());
163  return visited;
164 }

166 std::vector<Equation> Circuit::getEquations() {
167  std::vector<Equation> equations;
168 
169  auto refNodeIds = getRefNodes();
170 
171  for(auto r : refNodeIds){
172  // Reference nodes have a voltage of zero as the voltage is relative to this node.
173  equations.push_back(Equation(0, {
174  Term(1, new UnknownVoltage(r))
175  }));
176  }
177 
178  for(auto n : nodes){
179  //Create an equation containing the total current
180  // source multiplied by each of the outgoing and incoming
181  // current nodes from the node.
182 
183  // If n not in refNodeIds.
184  if(not std::count(refNodeIds.begin(), refNodeIds.end(), n)){
185  // All current nodes entering node n at n1.
186  auto incoming = getCurrents(n, CUR_IN);
187  // All current nodes leaving node n at n0.
188  auto outgoing = getCurrents(n, CUR_OUT);
189 
190  //Currents leave at n0 and enter at n1 because conventional
191  // currents are opposite to actual charge flow.
192 
193  // Add together the two lists of terms.
194  std::vector<Term> conserved;
195  conserved.insert(conserved.end(), incoming.begin(), incoming.end());
196  conserved.insert(conserved.end(), outgoing.begin(), outgoing.end());
197 
198  // Create an equation of total current equals the sum of all the terms.
199  equations.push_back(Equation(0, conserved));
200  }
201  }
202 
203  for(auto b : batteries){
204  // Make an equation for every battery.
205  equations.push_back(Equation(b.value, {
206  // N0 is negative as voltage is lost.
207  // N1 is the positive as voltage is gained.
208  Term(-1, new UnknownVoltage(b.n0)),
209  Term(1, new UnknownVoltage(b.n1))
210  }));
211  }
212 
213  return equations;
214 }

227 Solution Circuit::solve() {
228  auto equations = getEquations();
229  auto unknownCurrents = getUnknownCurrents();
230  std::vector<UnknownVoltage*> unknownVoltages;
231 
232  // Create an UnknownVoltage for each node as we don't know any voltage for any node.
233  for(auto v : nodes){
234  unknownVoltages.push_back(new UnknownVoltage(v));
235  }
236 
237  std::vector<Unknown*> unknowns(unknownCurrents.begin(), unknownCurrents.end());
238 
239  unknowns.insert(unknowns.end(), unknownVoltages.begin(), unknownVoltages.end());
240 
241  //Make two matrices, one which is the width of the number of equations by the number
242  // of variables, and the other has a width of the number of equations but is only 1 wide
243  auto A = Eigen::MatrixXd(equations.size(), getNumVars()).setZero();
244  auto z = Eigen::MatrixXd(equations.size(), 1).setZero();
245 
246  for(int i=0 ; i<(int)equations.size() ; i++){
247  // Apply the correct numbers of every equation onto the matrices,
248  // passing in a lambda to find the index of each
249  // term of the equation in class attribute unknowns.
250  equations.at(i).apply(i, &A, &z, [this, unknowns](Unknown* u) {
251  return getComponentIndex<Unknown>(unknowns, u);
252  });
253  }
254 
255  // Solve the matrix equation a = z
256  Eigen::MatrixXd x = A.fullPivLu().solve(z); // https://eigen.tuxfamily.org/dox/group__TutorialLinearAlgebra.html
257 
258  // A map of which nodes have which voltages on the circuit.
259  std::map<int, double> voltageMap;
260 
261  for(auto v : unknownVoltages){
262  // Get the voltage out from the solved matrix at the index of the component
263  // in the vector of unknowns. As the index is the node, and the node has the voltage.
264  auto voltage = x(getComponentIndex<Unknown>(unknowns, v));
265 
266  // Add to the voltage map
267  voltageMap.insert(std::pair<int, double>(v->node, voltage));
268  }
269 
270  //
271  std::vector<Component> elems;
272 
273  for(auto c : unknownCurrents){
274  // Set the new current for each component, as the matrix column 0 has the solved currents.
275  c->component.currentSolution = x(getComponentIndex<Unknown>(unknowns, c), 0);
276  elems.push_back(c->component);
277 
278  }
279 
280  return Solution(voltageMap, elems);
281 }
