17bool Solution::equals(Solution mnaSolution) {
18 std::vector<int> nodes; // The nodes of this circuit.
19 std::vector<int> otherNodes; // The nodes of the circuit to compare to.
20
21 for(auto n : voltageMap){
22 nodes.push_back(n.first);
23 }
24
25 for(auto n : mnaSolution.voltageMap){
26 otherNodes.push_back(n.first);
27 }
28
29 for(auto key : nodes){
30 // Check if every key in this equals the same key in the solution.
31 if(!numApproxEquals(getNodeVoltage(key), mnaSolution.getNodeVoltage(key))){
32 return false;
33 }
34 }
35
36 // Check if all the currents in this are in solution.
37 if((!hasAllComponents(mnaSolution))){
38 return false;

40
41 // Check if all the currents in solution are in this.
42 if(!mnaSolution.hasAllComponents(*this)) {
43 return false;
44 }

47 return true;
48}

92bool Solution::numApproxEquals(double a, double b) {
93 // Finds the difference between a and b then compares the difference
94 // to an epsilon.
95 return fabs(a - b) <= ( (fabs(a) < fabs(b) ? fabs(b) : fabs(a)) * APPROX_EPSILON);
