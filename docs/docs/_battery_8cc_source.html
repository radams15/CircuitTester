9Battery::Battery(double voltage, bool on) : UIComponent(ID, ":/images/battery.png") {

18 voltageBox->addWidget(voltageLabel);

20 // Add the voltage box to the main settings layout.

23 // Box to hold on/off button and label
24 auto* onOffBox = new QHBoxLayout;

26 onOffCheckbox = new QCheckBox;
27 // Add components to on/off box.
28 onOffBox->addWidget(onOffLabel);
29 onOffBox->addWidget(onOffCheckbox);
30 // Add on/off box to the main settings layout.
31 settingsBox->addLayout(onOffBox);
32
33 // Validate voltage to ensure that it is greater than 0.1V. If it is less than 0.1V set voltage to 0.1V.
34 voltageSpinner->setValue(voltage < 0.1? 0.1 : voltage);
35 // Can't validate a bool.
36 onOffCheckbox->setChecked(on);
37
38 // Add spacer at bottom to push all widgets up to top.
