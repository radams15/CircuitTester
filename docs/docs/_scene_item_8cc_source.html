71QPointF SceneItem::centrePoint() {
72 // Get width and height of the image.
73 double w = pixmap.width();
74 double h = pixmap.height();
75
76 QPointF p = pos();
77
78 // Get x, y coordinates of this SceneItem.
79 double x = p.x();

81
82 // Find the centre of the SceneItem by adding half the width to the x and adding half the height to the y.

125void SceneItem::setColour(QColor colour) {
126 QImage tmp = pixmap.toImage();
127
128 // Iterate over every pixel in the image.
129 for(int y1 = 0; y1 < tmp.height(); y1++) {
130 for (int x1 = 0; x1 < tmp.width(); x1++) {
131
132 // If the pixel is not completely transparent.
133 if(tmp.pixelColor(x1, y1).alpha() > 0) {
134
135 colour.setAlpha(tmp.pixelColor(x1,y1).alpha());
136
137 // Set the pixel to the new colour.
138 tmp.setPixelColor(x1, y1, colour);
139 }
140 }
141 }
142
143 // Turn back into pixmap
144 pixmap = QPixmap::fromImage(tmp);
145
146 // Refresh the image that is drawn when moved.
147 setPixmap(pixmap);
148}
