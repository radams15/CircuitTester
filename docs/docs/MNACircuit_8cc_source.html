14 T MNACircuit::vecPopFront(std::vector<T>& vec){
15  std::reverse(vec.begin(),vec.end()); // first becomes last, reverses the vector
16  T out = vec.back(); // get last element
17  vec.pop_back(); // pop last element off the vector
18  std::reverse(vec.begin(),vec.end()); // reverses it again, so the elements are in the same order as before
19 
20  return out;

72 std::vector<Term *>* MNACircuit::getCurrents(int node, int side) {
73  /*Convert the side to the current direction - if side is 0, current direction
74  * is 1, otherwise the current direction is -1*/
75  int currentDirection = side == 0 ? 1 : -1;
76 
77  // Check that the side is valid - either 0 or 1.
78  if(side != 0 and side != 1) throw std::invalid_argument("Invalid Side!");
79 
80  // The current terms to return.
81  auto* out = new std::vector<Term*>;
82 
83  for(auto b : batteries){
84  // Convert node index into node value.
85  int n = side == 0? b->n0 : b->n1;
86 
87  // If this is the node that was specified in the parameter.
88  if(n == node) {
89  // Batteries have an unknown current.
90  out->push_back(new Term(currentDirection, new UnknownCurrent(b)));
91  }
92  }
93 
94  for(auto r : resistors) {
95  // Convert node index into node value.
96  int n = side == 0 ? r->n0 : r->n1;
97 
98  // If correct node
99  if(n == node){
100  // The voltage enters and leaves the component, so there are 2 terms. This uses V=IR to find current, e.g. (1/R)*V = V/R = I
101  out->push_back(new Term(-currentDirection / r->value, new UnknownVoltage(r->n1 )));
102  out->push_back(new Term(currentDirection / r->value, new UnknownVoltage(r->n0 )));
103  }
104  }
105 
106  return out;
107 }

110  auto* out = new std::vector<int>;
111 
112  std::vector<int> toVisit = nodes;
113 
114  while(! toVisit.empty()){
115  // Get the element to visit next from toVisit.
116  int refNodeId = toVisit.at(0);
117 
118  // Add the reference node to the output list
119  out->push_back(refNodeId);
120 
121  std::vector<int>* connectedNodes = getConnectedNodes(refNodeId);
122 
123  // Remove every node that is directly connected to this reference node,
124  // as there cannot be 2 nodes that are connected but are both reference nodes.
125  for(auto c : *connectedNodes){
126  // Remove c from toVisit by iterating over every item and removing
127  // the matching one.
128  toVisit.erase(std::remove_if(toVisit.begin(), toVisit.end(), [c](int i){
129  return i == c;
130  }), toVisit.end());
131  }
132  }
133 
134  return out;

137 std::vector<int>* MNACircuit::getConnectedNodes(int node) {
138  // A queue of visited nodes from this node.
139  auto* visited = new std::vector<int>;
140 
141  // A list of nodes that we want to visit in the future.
142  std::vector<int> toVisit = {node};
143 
144  // This is basically a breadth first iteration of every element connected to this node.
145  while(not toVisit.empty()){
146  // Get the top vector item.
147  int nodeToVisit = vecPopFront<int>(toVisit);
148 
149  visited->push_back(nodeToVisit);
150 
151  for(auto e : elements){
152  if(e->contains(nodeToVisit)){
153  // Get the node leaving the element.
154  int oppositeNode = e->opposite(nodeToVisit);
155 
156  // Visit the opposite node in the future if we have not already visited it.
157  if(! std::count(visited->begin(), visited->end(), oppositeNode)){
158  toVisit.push_back(oppositeNode);
159  }
160  }
161  }
162  }
163 
164  // Remove duplicate values, just in case there was an error.
165  visited->erase(unique( visited->begin(), visited->end() ), visited->end());
166  return visited;
167 }

169 std::vector<Equation *>* MNACircuit::getEquations() {
170  auto* equations = new std::vector<Equation *>;
171 
172  std::vector<int>* refNodeIds = getRefNodes();
173 
174  for(auto r : *refNodeIds){
175  // Reference nodes have a voltage of zero as the voltage is relative to this node.
176  equations->push_back(new Equation(0, {
177  new Term(1, new UnknownVoltage(r))
178  }));
179  }
180 
181  for(auto n : nodes){
182  //Create an equation containing the total current
183  // source multiplied by each of the outgoing and incoming
184  // current nodes from the node.
185 
186  // If n not in refNodeIds.
187  if(not std::count(refNodeIds->begin(), refNodeIds->end(), n)){
188  // All current nodes entering node n at n1.
189  std::vector<Term *>* incoming = getCurrents(n, CUR_IN);
190  // All current nodes leaving node n at n0.
191  std::vector<Term *>* outgoing = getCurrents(n, CUR_OUT);
192 
193  //Currents leave at n0 and enter at n1 because conventional
194  // currents are opposite to actual charge flow.
195 
196  // Add together the two lists of terms.
197  auto* conserved = new std::vector<Term *>;
198  conserved->insert(conserved->end(), incoming->begin(), incoming->end());
199  conserved->insert(conserved->end(), outgoing->begin(), outgoing->end());
200 
201  // Create an equation of total current equals the sum of all the terms.
202  equations->push_back(new Equation(0, *conserved));
203  }
204  }
205 
206  for(auto b : batteries){
207  // Make an equation for every battery.
208  equations->push_back(new Equation(b->value, {
209  // N0 is negative as voltage is lost.
210  // N1 is the positive as voltage is gained.
211  new Term(-1, new UnknownVoltage(b->n0)),
212  new Term(1, new UnknownVoltage(b->n1))
213  }));
214  }
215 
216  return equations;
217 }

230 MNASolution *MNACircuit::solve() {
231  std::vector<Equation *>* equations = getEquations();
232  std::vector<UnknownCurrent*>* unknownCurrents = getUnknownCurrents();
233  auto* unknownVoltages = new std::vector<UnknownVoltage*>;
234 
235  // Create an UnknownVoltage for each node as we don't know any voltage for any node.
236  for(auto v : nodes){
237  unknownVoltages->push_back(new UnknownVoltage(v));
238  }
239 
240  auto* unknowns = new std::vector<Unknown*>(unknownCurrents->begin(), unknownCurrents->end());
241 
242  unknowns->insert(unknowns->end(), unknownVoltages->begin(), unknownVoltages->end());
243 
244  //Make two matrices, one which is the width of the number of equations by the number
245  // of variables, and the other has a width of the number of equations but is only 1 wide
246  auto A = Eigen::MatrixXd(equations->size(), getNumVars()).setZero();
247  auto z = Eigen::MatrixXd(equations->size(), 1).setZero();
248 
249  for(int i=0 ; i<equations->size() ; i++){
250  // Apply the correct numbers of every equation onto the matrices,
251  // passing in a lambda to find the index of each
252  // term of the equation in class attribute unknowns.
253  equations->at(i)->apply(i, &A, &z, [this, unknowns](Unknown *u) {
254  return getElementIndex<Unknown>(unknowns, u);
255  });
256  }
257 
258  // Solve the matrix equation a = z
259  Eigen::MatrixXd x = A.fullPivLu().solve(z); // https://eigen.tuxfamily.org/dox/group__TutorialLinearAlgebra.html
260 
261  // A map of which nodes have which voltages on the circuit.
262  auto* voltageMap = new std::map<int, double>;
263 
264  for(auto v : *unknownVoltages){
265  // Get the voltage out from the solved matrix at the index of the element
266  // in the vector of unknowns. As the index is the node, and the node has the voltage.
267  auto voltage = x(getElementIndex<Unknown>(unknowns, v));
268 
269  // Add to the voltage map
270  voltageMap->insert(std::pair<int, double>(v->node, voltage));
271  }
272 
273  //
274  std::vector<MNAComponent*> elems;
275 
276  for(auto c : *unknownCurrents){
277  // Set the new current for each element, as the matrix column 0 has the solved currents.
278  c->element->currentSolution = x(getElementIndex<Unknown>(unknowns, c), 0);
279  elems.push_back(c->element);
280 
281  }
282 
283  return new MNASolution(*voltageMap, elems);
284 }
